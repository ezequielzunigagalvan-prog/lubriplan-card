// ← Estado nuevo
const [filtroEstado, setFiltroEstado] = useState('todos')
const [ultimasLubricaciones, setUltimasLubricaciones] = useState({})

// ← Effect para obtener últimas lubricaciones
useEffect(() => {
  equipos.forEach(eq => {
    eq.puntos?.forEach(punto => {
      obtenerUltimaLubricacion(punto.id)
        .then(res => {
          setUltimasLubricaciones(prev => ({
            ...prev,
            [punto.id]: res?.fecha
          }))
        })
    })
  })
}, [equipos])

// ← Función para filtrar
const equiposCumplesFiltro = (equipo) => {
  const hoy = new Date().toISOString().split('T')[0]
  const hace7dias = new Date(Date.now() - 7*24*60*60*1000)
    .toISOString().split('T')[0]
  
  // Obtener última lubricación del equipo
  const ultimaLub = Math.max(
    ...equipo.puntos.map(p => 
      ultimasLubricaciones[p.id] || '2000-01-01'
    )
  )
  
  switch (filtroEstado) {
    case 'hoy':
      return ultimaLub === hoy
    case 'semana':
      return ultimaLub >= hace7dias
    case 'pendientes':
      return ultimaLub < hace7dias || !ultimaLub
    case 'nunca':
      return !ultimaLub
    default:
      return true
  }
}

const equiposMostrados = equipos.filter(equiposCumplesFiltro)

// ← HTML del filtro
<div style={{ display: 'flex', gap: 8, marginBottom: 16, padding: '0 16px' }}>
  {['todos', 'hoy', 'semana', 'pendientes', 'nunca'].map(opcion => (
    <button
      key={opcion}
      onClick={() => setFiltroEstado(opcion)}
      style={{
        padding: '8px 14px',
        borderRadius: 20,
        border: 'none',
        background: filtroEstado === opcion ? '#6366f1' : BG_CARD,
        color: filtroEstado === opcion ? '#fff' : TEXT_SUB,
        cursor: 'pointer',
        fontSize: 12,
        fontWeight: 600,
      }}
    >
      {opcion === 'todos' ? '✓ Todos'
        : opcion === 'hoy' ? '🕐 Hoy'
        : opcion === 'semana' ? '📅 Esta semana'
        : opcion === 'pendientes' ? '🔴 Pendientes'
        : '⭕ Nunca'
      }
    </button>
  ))}
</div>
```

// server/routes/card.js
// Obtener puntos pendientes (no lubricados en últimos 7 días)
router.get('/reportes/pendientes', requireAuth, async (req, res) => {
  try {
    const hace7dias = new Date(Date.now() - 7*24*60*60*1000)
      .toISOString().split('T')[0]
    
    const { rows } = await pool.query(`
      SELECT 
        p.id, p.nombre as punto_nombre, p.frecuencia,
        e.id as equipo_id, e.nombre as equipo_nombre,
        COALESCE(MAX(lh.fecha), '2000-01-01') as ultima_lubricacion,
        EXTRACT(DAY FROM CURRENT_DATE - COALESCE(MAX(lh.fecha), '2000-01-01'))::int as dias_atraso
      FROM puntos_lubricacion_card p
      JOIN equipos_card e ON p.equipo_id = e.id
      LEFT JOIN lubricaciones_historial lh ON p.id = lh.punto_id
      GROUP BY p.id, p.nombre, p.frecuencia, e.id, e.nombre
      HAVING COALESCE(MAX(lh.fecha), '2000-01-01') < $1
      ORDER BY COALESCE(MAX(lh.fecha), '2000-01-01') ASC
    `, [hace7dias])
    
    res.json(rows.map(r => ({
      puntoId: r.id,
      puntoNombre: r.punto_nombre,
      equipoId: r.equipo_id,
      equipoNombre: r.equipo_nombre,
      frecuencia: r.frecuencia,
      ultimaLubricacion: r.ultima_lubricacion,
      diasAtraso: r.dias_atraso,
    })))
  } catch (err) {
    console.error('[GET pendientes]', err)
    res.status(500).json({ error: 'Error obteniendo pendientes' })
  }
})
```
